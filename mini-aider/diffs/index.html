+def wildcard_filter(files, pattern):
+    """
+    过滤出匹配通配符模式的文件列表。
+    支持 * (匹配任意字符序列) 和 ? (匹配单个任意字符)
+    
+    Args:
+        files: 字符串列表，表示文件名
+        pattern: 字符串，通配符模式，如 "*.py", "test?.txt"
+    
+    Returns:
+        匹配的文件名列表
+    """
+    import re
+    
+    # 将通配符模式转换为正则表达式
+    # 转义正则特殊字符（除了 * 和 ?）
+    escaped = re.escape(pattern)
+    # 将 \* 替换为 .*，\? 替换为 .
+    regex_pattern = escaped.replace('\\*', '.*').replace('\\?', '.')
+    # 添加行首行尾锚点
+    regex_pattern = '^' + regex_pattern + '$'
+    
+    compiled = re.compile(regex_pattern)
+    return [f for f in files if compiled.match(f)]
